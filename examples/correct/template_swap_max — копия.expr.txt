template <type Type>
void function Swap(Type a, Type b) {
    Type temp = a;
    a = b;
    b = temp;
}

template <type T>
T function Max(T a, T b) {
    if (a > b) then {
        return a;
    } else {
        return b;
    }
}

template <type ContainerType, type ElementType>
void function PrintContainer(ContainerType container, ElementType element) {
    out(element);
}

template <type T>
T function GetDefault() {
    return (T)0;
}

int function Main() {
    int x = 5;
    int y = 10;
    
    out("Swap");
    out(x);
    out(y);
    Swap<int>(x, y);
    out(x);
    out(y);
    
    out("maxVal");
    int maxVal = Max<int>(x, y);
    out(maxVal);
    

    int defaultInt = GetDefault<int>();
    float defaultFloat = GetDefault<float>();
    
    out("Default int");
    out(defaultInt);

    out("Default float");
    out(defaultFloat);
    
    char a = 'A';
    char b = 'B';

    out("Swap");
    out(a);
    out(b);

    Swap<char>(a, b);
    out(a);
    out(b);
    
    char maxChar = Max<char>(a, b);
    out(maxChar);
    
    return 0;
}
